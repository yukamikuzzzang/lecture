select to_char(sysdate,'day') from dual; 	수요일/Wed..
select to_char(sysdate,'mon') from dual; 	3월 / Mar

'80/12/24' 문자형에서 년도만 뽑아내라.
select to_char(to_date('80/12/24'),'yyyy') from dual; 	       (밑에 결과와 같음)
select to_char(to_date('80/12/24','rr/mm/dd'),'yyyy') from dual;  (rr = 50년이상은 19--)
select to_char(to_date('80/12/24','yy/mm/dd'),'yyyy') from dual; (yy = 20--)
